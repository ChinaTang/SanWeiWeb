CREATE TABLE user_info (id INTEGER UNSIGNED AUTO_INCREMENT PRIMARY KEY, USER_NAME VARCHAR (100) NOT NULL, PASS_WORD VARCHAR(200) NOT NULL, TEACHER_ID VARCHAR(200) NOT NULL, REMARKS VARCHAR(100) NOT NULL);

CREATE TABLE teach_info (id INTEGER UNSIGNED AUTO_INCREMENT PRIMARY KEY, TEACHER_ID VARCHAR(200) NOT NULL, TEACHER_NAME VARCHAR(200) NOT NULL, WORK_STATE VARCHAR(200) NOT NULL, CURRICULUM VARCHAR(200) NOT NULL, REMARKS VARCHAR(100) NOT NULL);

CREATE TABLE students_info (id INTEGER UNSIGNED AUTO_INCREMENT PRIMARY KEY,START_SEMESTER DATE NOT NULL, OVER_SENESTER DATE NOT NULL,  TEACHHER_ID VARCHAR(200) NOT NULL, STUDENT_NAME VARCHAR(200) NOT NULL,GUARDIAN_NAME VARCHAR(200) NOT NULL, SCHOOL_NAME VARCHAR(200) NOT NULL, TEL_PHONE VARCHAR(200) NOT NULL, ADDRESS VARCHAR(400) NOT NULL, REMARKS VARCHAR(100) NOT NULL);

CREATE TABLE achievement(id INTEGER UNSIGNED AUTO_INCREMENT PRIMARY KEY, START_SEMESTER DATE NOT NULL, OVER_SEMESTER DATE NOT NULL, SUBJECT_NAME VARCHAR(100) NOT NULL, MARK VARCHAR(50) NOT NULL, TEACHER_ID VARCHAR(200) NOT NULL, STUDENT_NAME VARCHAR(200) NOT NULL, REMARKS VARCHAR(200) NOT NULL);

CREATE TABLE entered_continuation(id INTEGER UNSIGNED AUTO_INCREMENT PRIMARY KEY, START_SEMESTER DATE NOT NULL, OVER_SEMESTER DATE NOT NULL, STUDENT_NAME VARCHAR(100) NOT NULL, TEACHER_ID VARCHAR(200) NOT NULL, REMARKS VARCHAR(100) NOT NULL);
